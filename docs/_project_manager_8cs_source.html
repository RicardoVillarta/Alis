293  private void CreateProject(string name, string directory, string mode)
294  {
295  if (!Directory.Exists(directoryField + "/" + nameField))
296  {
297  List<Project> temp = LocalData.Load<List<Project>>("Projects");
298 
299  if (temp == null)
300  {
301  temp = new List<Project>();
302  }
303 
304  if (!temp.Any(i => (i.Directory + "/" + i.Name).Equals(directory + "/" + name)))
305  {
306  string dir = directory + "/" + name;
307  string assetPath = directory + "/" + name + "/Assets";
308  string configPath = directory + "/" + name + "/Config";
309  string dataPath = directory + "/" + name + "/Data";
310  string libPath = directory + "/" + name + "/Lib";
311 
312  Project project = new Project(name, dir, assetPath, configPath, dataPath, libPath);
313  Logger.Warning("Project: " + name + " at " + dir);
314 
315  projects.Add(project);
316  LocalData.Save<List<Project>>("Projects", projects);
317 
318  Directory.CreateDirectory(dir);
319  Directory.CreateDirectory(assetPath);
320  Directory.CreateDirectory(configPath);
321  Directory.CreateDirectory(dataPath);
322  Directory.CreateDirectory(libPath);
323 
324  string projectFile = File.ReadAllText(Application.ProjectPath + "/Resources/DefaultPr.txt", Encoding.UTF8);
325  File.WriteAllText(dir + "/" + name + ".csproj", projectFile, Encoding.UTF8);
326 
327  string solutionFile = File.ReadAllText(Application.ProjectPath + "/Resources/DefaultSl.txt", Encoding.UTF8).Replace("Example", name);
328  File.WriteAllText(dir + "/" + name + ".sln", solutionFile, Encoding.UTF8);
329 
330  string program = File.ReadAllText(Application.ProjectPath + "/Resources/Program.txt", Encoding.UTF8);
331  File.WriteAllText(dir + "/" + "Program" + ".cs", program, Encoding.UTF8);
332 
333 
334  File.Copy(Application.ProjectPath + "/Resources/Core.dll", libPath + "/" + "Core" + ".dll");
335  File.Copy(Application.ProjectPath + "/Resources/Tools.dll", libPath + "/" + "Tools" + ".dll");
336 
337  VideoGame game = new VideoGame(new Config(name), new Scene("Default"));
338  LocalData.Save<VideoGame>("Data", dataPath, game);
339 
340  Project.ChangeProject(project, game);
341 
342  LocalData.Save<Project>(name, dir, project);
343 
344  Close();
345  }
346  }

372  private void DeleteProject(Project project)
