198  private void Update()
199  {
200  Debug.Log("\nRunning Alis...");
201 
202  while (window.Exists)
203  {
204  snapshot = window.PumpEvents();
205  if (!window.Exists)
206  {
207  Debug.Log("Windows not Exists.");
208  break;
209  }
210 
211  controller.Update(deltaSeconds, snapshot);
212  widgetManager.Update();
213  ImGui.ShowDemoWindow();
214  ShowPreviewMessage();
215 
216  commandList.Begin();
217  commandList.SetFramebuffer(graphicsDevice.MainSwapchain.Framebuffer);
218  commandList.ClearColorTarget(0, clearColor);
219  controller.Render(graphicsDevice, commandList);
220  commandList.End();
221  graphicsDevice.SubmitCommands(commandList);
222  graphicsDevice.SwapBuffers(graphicsDevice.MainSwapchain);
223  }
224  }
