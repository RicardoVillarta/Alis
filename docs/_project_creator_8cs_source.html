274  private void CreateProject(string name, string directory, string current_item)
275  {
276  if (Directory.Exists(directory + "/" + name))
277  {
278  Debug.Error("The project(" + directory + "/" + name + ") already exists.");
279  messageError = true;
280  }
281  else
282  {
283  Directory.CreateDirectory(directory + "/" + name);
284  Directory.CreateDirectory(directory + "/" + name + "/Assets");
285  Directory.CreateDirectory(directory + "/" + name + "/Config");
286 
287  string content = File.ReadAllText(Application.ProjectPath + "/Resources/DefaultPr.txt", Encoding.UTF8);
288  File.WriteAllText(directory + "/" + name + "/" + name + ".csproj", content, Encoding.UTF8);
289 
290  content = File.ReadAllText(Application.ProjectPath + "/Resources/DefaultSl.txt", Encoding.UTF8);
291  content = content.Replace("Example", name);
292  File.WriteAllText(directory + "/" + name + "/" + name + ".sln", content, Encoding.UTF8);
293 
294  Project.AssetsPath = directory + "/" + name + "/Assets";
295  Project.Name = name;
296  Project.CurrentPath = directory + "/" + name;
297  Project.ConfigPath = directory + "/" + name + "/Config";
298 
299  Project project = new Project(name, directory + "/" + name, directory + "/" + name + "/Assets", directory + "/" + name + "/Config");
300 
301  List<Project> projects = LocalData.Load<List<Project>>("Projects");
302  if (projects == null)
303  {
304  projects = new List<Project>();
305  }
306 
307  projects.Add(project);
308 
309  LocalData.Save<List<Project>>("Projects", projects);
310 
311  Project.ChangeProject();
312 
313  Close();
314  }
