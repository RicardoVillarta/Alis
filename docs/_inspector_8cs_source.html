94  private void SeeObjComponents(GameObject gameObject)
95  {
96  ImGui.BeginGroup();
97  ImGui.BeginChild("GameObject-Child", new Vector2(ImGui.GetContentRegionAvail().X, 80.0f), true);
98 
99  string content = gameObject.Name;
100 
101  ImGui.Text("Name: ");
102 
103  ImGui.PushItemWidth(ImGui.GetContentRegionAvail().X - 25.0f);
104 
105  if (ImGui.InputText(Icon.CUBE + " ##" + gameObject.Name, ref content, 512, ImGuiInputTextFlags.EnterReturnsTrue))
106  {
107  gameObject.Name = content;
108  }
109 
110  ImGui.PopItemWidth();
111 
112 
113  ImGui.EndChild();
114  ImGui.EndGroup();
115 
116 
117  ImGui.BeginGroup();
118  ImGui.AlignTextToFramePadding();
119  if (ImGui.TreeNodeEx(gameObject.Transform.Icon + " " + gameObject.Transform.GetType().Name, ImGuiTreeNodeFlags.AllowItemOverlap))
120  {
121  foreach (PropertyInfo property in gameObject.Transform.GetType().GetProperties())
122  {
123  foreach (KeyValuePair<Type, Action<IComponent, PropertyInfo>> field in fields)
124  {
125  if (field.Key.Equals(property.PropertyType) && property.CanWrite)
126  {
127  field.Value.Invoke((IComponent)gameObject.Transform, property);
128  }
129  }
130  }
131 
132  ImGui.TreePop();
133  }
134 
135  ImGui.EndGroup();
136 
137  foreach (IComponent component in gameObject.Components)
138  {
139  ImGui.BeginGroup();
140  ImGui.AlignTextToFramePadding();
141  if (ImGui.TreeNodeEx(component.GetType().GetProperties().ToList().Find(i => i.Name.Equals("Icon")).GetValue(component) + " " + component.GetType().Name, ImGuiTreeNodeFlags.AllowItemOverlap))
142  {
143  foreach (PropertyInfo property in component.GetType().GetProperties())
144  {
145  foreach (KeyValuePair<Type, Action<IComponent, PropertyInfo>> field in fields)
146  {
147  if (field.Key.Equals(property.PropertyType) && property.CanWrite)
148  {
149  field.Value.Invoke(component, property);
150  }
151  }
152  }
153 
154  ImGui.TreePop();
155  }
156 
157  ImGui.EndGroup();
158  }
159 
160  if (ImGui.Button("Add Component", new Vector2(ImGui.GetContentRegionAvail().X, 30f)))
161  {
162  ImGui.OpenPopup("ElementList");
163  }
164 
165  if (ImGui.BeginPopup("ElementList"))
166  {
167 
168  Type type = typeof(IComponent);
169  IEnumerable<Type> types = AppDomain.CurrentDomain.GetAssemblies()
170  .SelectMany(s => s.GetTypes())
171  .Where(p => type.IsAssignableFrom(p));
172 
173  foreach (Type component in types)
174  {
175  if (!component.Name.Equals("IComponent") && !component.Name.Equals("Transform"))
176  {
177  if (ImGui.MenuItem(component.Name))
178  {
179  constructors[component].Invoke(gameObject);
180  }
181  }
182  }
183  }
