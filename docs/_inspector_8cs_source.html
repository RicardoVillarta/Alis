74  ImGui.AlignTextToFramePadding();
75 
76  foreach (var propertyInfo in gameObject.Transform.GetType().GetProperties())
77  {
78  if (propertyInfo.PropertyType.Equals(typeof(Vector3)))
79  {
80  ImGui.Text(propertyInfo.Name);
81  ImGui.Columns(3, propertyInfo.Name, true);
82  Vector3 content = (Vector3)propertyInfo.GetValue(gameObject.Transform);
83  float x = content.X;
84 
85  if (ImGui.InputFloat("X" + "## " + propertyInfo.Name, ref x))
86  {
87  content.X = x;
88  propertyInfo.SetValue(gameObject.Transform, content);
89  }
90 
91  ImGui.NextColumn();
92 
93  float y = content.Y;
94  if (ImGui.InputFloat("Y" + "## " + propertyInfo.Name, ref y))
95  {
96  content.Y = y;
97  propertyInfo.SetValue(gameObject.Transform, content);
98  }
99 
100  ImGui.NextColumn();
101 
102  float z = content.Z;
103  if (ImGui.InputFloat("Z" + "## " + propertyInfo.Name, ref z))
104  {
105  content.Z = z;
106  propertyInfo.SetValue(gameObject.Transform, content);
107  }
108  ImGui.Columns(1);
109  }
110 
111  }
112 
113 
114  foreach (IComponent component in gameObject.Components)
115  {
116  if (ImGui.TreeNodeEx(component.GetType().Name, ImGuiTreeNodeFlags.AllowItemOverlap))
117  {
118  foreach (var propertyInfo in component.GetType().GetProperties())
119  {
120  if (propertyInfo.PropertyType.Equals(typeof(string)))
121  {
122  string content = (string)propertyInfo.GetValue(component) ?? "";
123 
124  if (ImGui.InputText(propertyInfo.Name, ref content, 256, ImGuiInputTextFlags.EnterReturnsTrue))
125  {
126  propertyInfo.SetValue(component, content);
127  }
128  }
129 
130  if (propertyInfo.PropertyType.Equals(typeof(int)))
131  {
132  int content = (int)propertyInfo.GetValue(component);
133  if (ImGui.InputInt(propertyInfo.Name, ref content, 1))
134  {
135  propertyInfo.SetValue(component, content);
136  }
137  }
138 
139  if (propertyInfo.PropertyType.Equals(typeof(float)))
140  {
141  float content = (float)propertyInfo.GetValue(component);
142  if (ImGui.InputFloat(propertyInfo.Name, ref content, 0.1f))
143  {
144  propertyInfo.SetValue(component, content);
145  }
146  }
147 
148  if (propertyInfo.PropertyType.Equals(typeof(bool)))
149  {
150  bool content = (bool)propertyInfo.GetValue(component);
151  if (ImGui.Checkbox(propertyInfo.Name, ref content))
152  {
153  propertyInfo.SetValue(component, content);
154  }
155  }
156 
157  if (propertyInfo.PropertyType.Equals(typeof(List<string>)))
158  {
159  List<string> content = (List<string>)propertyInfo.GetValue(component);
160  foreach (string row in content)
161  {
162  string rowContent = row;
163  if (ImGui.InputText(row + "## " + content.IndexOf(row), ref rowContent, 256, ImGuiInputTextFlags.EnterReturnsTrue))
164  {
165  content[content.IndexOf(row)] = rowContent;
166  propertyInfo.SetValue(component, content);
167  }
168  }
169  }
170 
171 
172  if (propertyInfo.PropertyType.Equals(typeof(Vector2)))
173  {
174  Vector2 content = (Vector2)propertyInfo.GetValue(component);
175  if (ImGui.SliderFloat("X", ref content.X, -float.MaxValue, float.MaxValue))
176  {
177  propertyInfo.SetValue(component, content);
178  }
179 
180  ImGui.NextColumn();
181 
182  if (ImGui.SliderFloat("Y", ref content.Y, -float.MaxValue, float.MaxValue))
183  {
184  propertyInfo.SetValue(component, content);
185  }
186  }
187 
188  if (propertyInfo.PropertyType.Equals(typeof(Vector3)))
189  {
190  ImGui.Text(propertyInfo.Name);
191  ImGui.Columns(3, propertyInfo.Name, true);
192  Vector3 content = (Vector3)propertyInfo.GetValue(component);
