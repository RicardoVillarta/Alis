63  {
64  Render.Current.AddCollision(rectangle);
65  }
66 
67  rectangle.Position = new Vector2f(transform.Position.X, transform.Position.Y);
68  foreach(RectangleShape rectangleShape in Render.Current.Collisions)
69  {
70  if (!rectangleShape.Equals(rectangle))
71  {
72  var leftTop = new Vector2f(rectangle.GetGlobalBounds().Left, rectangle.GetGlobalBounds().Top);
73  var rightTop = new Vector2f(rectangle.GetGlobalBounds().Left + rectangle.GetGlobalBounds().Width, rectangle.GetGlobalBounds().Top);
74  var downLeft = new Vector2f(rectangle.GetGlobalBounds().Left, rectangle.GetGlobalBounds().Top + rectangle.GetGlobalBounds().Height);
75  var downRight = new Vector2f(rectangle.GetGlobalBounds().Left + rectangle.GetGlobalBounds().Width, rectangle.GetGlobalBounds().Top + rectangle.GetGlobalBounds().Height);
76 
77  var midleTop = new Vector2f(((rightTop.X - leftTop.X) / 2) + leftTop.X, leftTop.Y);
78  var midleDown = new Vector2f(((downRight.X - downLeft.X) / 2) + downLeft.X, downLeft.Y);
79  var midleLeft = new Vector2f(downLeft.X, ((downLeft.Y - leftTop.Y) / 2) + leftTop.Y);
80  var midleRight = new Vector2f(downRight.X, ((downRight.Y - rightTop.Y) / 2) + rightTop.Y);
81 
82 
83  if (Render.Current.Collisions.Find(i => (i != rectangle) && i.GetGlobalBounds().Intersects(rectangle.GetGlobalBounds())) != null)
84  {
85  GetGameObject().Components.ForEach(i => i.OnCollionStay(this));
86 
87  if (!isTrigger)
88  {
89  if (rectangleShape.GetGlobalBounds().Contains(midleTop.X, midleTop.Y))
90  {
91  Console.WriteLine("Chocando la parte de arriba de " + this.GetGameObject().Name);
92  this.GetGameObject().Transform.CanGoUp = false;
93  }
94 
95  if (rectangleShape.GetGlobalBounds().Contains(midleDown.X, midleDown.Y))
96  {
97  Console.WriteLine("Chocando la parte de abajo de " + this.GetGameObject().Name);
98  this.GetGameObject().Transform.CanGoDown = false;
99  }
100 
101  if (rectangleShape.GetGlobalBounds().Contains(midleLeft.X, midleLeft.Y))
102  {
103  Console.WriteLine("Chocando la parte izquierda de " + this.GetGameObject().Name);
104  this.GetGameObject().Transform.CanGoLeft = false;
105  }
106 
107  if (rectangleShape.GetGlobalBounds().Contains(midleRight.X, midleRight.Y))
108  {
109  Console.WriteLine("Chocando la parte derecha de " + this.GetGameObject().Name);
110  this.GetGameObject().Transform.CanGoRight = false;
111  }
112  }
113  }
114  else
115  {
116  if (!rectangleShape.GetGlobalBounds().Contains(midleTop.X, midleTop.Y))
117  {
118  this.GetGameObject().Transform.CanGoUp = true;
119  }
120 
121  if (!rectangleShape.GetGlobalBounds().Contains(midleDown.X, midleDown.Y))
122  {
123  this.GetGameObject().Transform.CanGoDown = true;
124  }
125

127  {
128  this.GetGameObject().Transform.CanGoLeft = true;
129  }
130
